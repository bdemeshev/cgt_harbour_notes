\begin{enumerate} 
    \item[2.1.] Using hockey stick decomposition or otherwise find the sums
    \[
    \sum_{k=1}^{n} {6k^2 + 4k}.
    \]

    \item[2.2] Take a deep breath and calculate 
\[
{-1 \choose 4},\quad  {4/3 \choose 3}, \quad  {-1/2 \choose 4}. 
\]


    \item[3.1.] Using Gauss summation trick or otherwise find the sum 
    \[
    -3{n\choose 0} -1 {n \choose 1} +1 \cdot {n \choose 2} + 3 \cdot {n \choose 3} + \dots + (2n - 3) {n \choose n}.
    \]
    \item[3.2.] Consider the equation $a + b + c = 11$ with integers $a \geq 1$, $b \geq 2$, $c\geq 3$.

    \begin{enumerate}
        \item How many solutions are there?
        \item Which solution would you encode as $\star \star \mid \, \mid \star\star\star$?
    \end{enumerate}

    \item[4.1.] Solve the recurrence equation:
    \[
    b_n = \frac{n + 3}{n + 4} b_{n-1}, \quad b_1 = 10.
    \]
    \item[4.2.] Here $L$ is the lag operator and $x_n = 5^n$. 
    Find the values
    \[
    (1 - 5L) x_n, \quad (1 - 2L)x_n, \quad (1-5L)(1 - 3L)x_n.
    \]


    \item[5.1.] Consider the recurrence equation $y_n - 8 y_{n-1} + 15 y_{n-2} = 0$.
    
    Find all solutions of the form $y_n = \lambda^n$.
    \item[5.2.] You know that $r_n = c \cdot 2^n + d \cdot n \cdot 3^n + n$ is the solution of some recurrence equation with initial conditions  $r_0 = 3$ and $r_1 = 10$.
    
    Find $c$ and $d$. 

    \item[6.1.] Calculate the following complex values: 
    \[
    (6 - 3i) \cdot (2 + 5i), \quad \frac{1 - 8i}{1 + 4i}, \quad (1 - 4i)^3.
    \]
    \item[6.2.] Consider the recurrence equation
    \[
    a_n - 4 a_{n-1} + 5 a_{n-2} = 4 n -14.
    \] 
    Find the solution of the form $a_n = \alpha n + \beta$.


    \item[7.1.] Extract the sequence $(a_n)$ from the generating functions:
    \[
    g(t) = \frac{1}{1 - 5t}, \quad g(t) = \frac{1}{1 + 2t}, \quad g(t) = \frac{t}{1 - 3t}.
    \]
    \item[7.2.] Decompose the fraction into a sum of more simple fractions
    \[
    g(t) = \frac{5 - 5t}{(1 + 2t)(1-3t)}.
    \]
\end{enumerate}
