\subsection*{Home assignment: day 2 $\to$ day 5}

Deadline: 2026-07-03, 16:59. 

\begin{enumerate}
    \item There are $10$ standard indistinguishable dices. 

\begin{enumerate}[label=\alph*)]
    \item How many different combinations of scores are possible?
    \item How many different combinations of scores are possible if at least one dice shows $6$?
    \item How many different combinations of scores are possible if no dice shows $5$ nor $6$?
\end{enumerate}

\item There are $5$ types of post-cards sold at the post-office. 
\begin{enumerate}[label=\alph*)]
\item In how many different ways can I buy $12$ post-cards?
\item In how many different ways can I buy $12$ post-cards if I would like to buy at least one post-card of every type?
\end{enumerate}

\item Using Gauss summation trick or otherwise find the sum:
\[
{n \choose 0} - 2 {n \choose 1} + 3 {n \choose 2} - 4 {n \choose 3} \dots + (-1)^n (n+1){n \choose n};
\]

\item Recall the mother-identity
\[
(1 + x)^n = 1 + {n \choose 1} x^1 + {n \choose 2} x^2 + \dots + {n \choose n} x^n.
\]

\begin{enumerate}
    \item Find the first derivative of the mother-identity. 
    \item Find the value of the sum
\[
{n \choose 1} + 2 \cdot 7 \cdot {n \choose 2} + 3 \cdot 7^2 \cdot {n \choose 2} + \dots + n \cdot 7^{n-1} {n \choose n}.
\]
\item Find the second derivative of the mother-identity.
\item Find the value of the sum
\[
2 \cdot 1 {n \choose 2} + 3 \cdot 2 \cdot 9 \cdot {n \choose 3} + 4 \cdot 3 \cdot 9^2 \cdot {n \choose 4} + 5 \cdot 4 \cdot 9^3 \cdot {n \choose 5} + \dots + n(n-1) 9^{n-2} {n \choose n}.
\]

\end{enumerate}

\end{enumerate}
