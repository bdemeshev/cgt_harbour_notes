\section{Day 1. Classic binomial and multinomial identities}

\subsection*{Planned entrance quiz :)}

\begin{enumerate}
\item How many ways are to order $6$ persons in a queue?
\item The box contains $50$ balls numbered from $1$ to $50$. We draw $6$ balls without replacement and write down the resulting sequence of numbers.
How many different sequences are possible?
\item I have $5$ yellow books and $7$ red books. I want to place them on my bookshelf and I would like to place yellow books to the left of red books. 
How many book orders are possible?
\end{enumerate}


\begin{definition}
    Factorial of $n$, denoted by $n!$, is the number of ways to put $n$ distinct objects in a queue. 
\end{definition}


\begin{definition}
    Binomial coefficient $n \choose k$ is the number of ways to select $k$ distinct objects from $n$ distinct objects.
\end{definition}

\begin{definition}
    For $n = a + b + c$ the multinomial coefficient $n \choose a, b, c$ is the number of ways to split $n$ distinct objects into different boxes, 
    where the first box contains $a$ objects, the second box contains $b$ objects, the third box contains $c$ objects. 

    One may create multinomial coefficient for any number of boxes :)
\end{definition}


\begin{exercise}
Is $n \choose k$ equal to $n \choose n-k$? 

Is $n \choose k$ equal to $n \choose k, n-k$?
\end{exercise}

\begin{exercise}
Write the product
    \[
    {100 \choose 20} \cdot {80 \choose 50} \cdot {30 \choose 10}
    \]
as one multinomial coefficient.
\end{exercise}

\begin{exercise}
Write 
\[
n \choose a, b, c, d
\] using binomial coefficients.

Write 
\[
n \choose a, b, c, d
\]
using factorials of $n$, $a$, $b$, $c$ and $d$.
\end{exercise}

\begin{idea}
Solution strategies for combinatorial identities:
\begin{enumerate}
    \item Double counting or story proof. 
    \item Visual identity (rare gem!).
    \item Evaluation of a polynomial at a special point. Differentiation, integration.
    \item Combine old identities into new. 
    \item Guess the answer and verify by induction. 
    \item Dark art!
\end{enumerate}
\end{idea}

\subsection*{Stars and bars}

\begin{exercise}
There are $20$ chairs in a row and $7$ students. Each student needs one chair to sit down. 
We do care only about which chairs are occupied.

\begin{enumerate}[label=\alph*)]
    \item In how many ways chairs can be occupied? 
    \item In how many ways chairs can be occupied if there should be at least one empty chair between occupied chairs?
    \item In how many ways chairs can be occupied if there should be at least two empty chairs between occupied chairs?
\end{enumerate}
\end{exercise}

\begin{exercise}
I have four bags of different colors and $20$ identical marble stones. 

\begin{enumerate}[label=\alph*)]
    \item In how many different ways can I put marbles into bags?
    \item In how many different ways can I put marbles into bags if there should be no empty bags?
\end{enumerate}
\end{exercise}

\begin{theorem}[Stars and bars theorem]
The number of ways to distribute $n$ identical objects into $k$ distinguishable bags is equal to
\[
{n + k - 1 \choose n} = {n + k - 1 \choose k - 1}.
\]
Hint: $n$ is the number of stars and $(k - 1)$ is the number of bars. 
\end{theorem}

\begin{exercise}
There are $5$ types of post-cards sold at the post-office. 
\begin{enumerate}[label=\alph*)]
\item In how many different ways can I buy $12$ post-cards?
\item In how many different ways can I buy $12$ post-cards if I would like to buy at least one post-card of every type?
\end{enumerate}
\end{exercise}

\begin{exercise}
Consider the equation $x_1 + x_2 + x_3 + x_4 = 2026$.

\begin{enumerate}[label=\alph*)]
\item How many solutions does it have if $x_1$, $x_2$, $x_3$ and $x_4$ are non-negative integers?
\item How many solutions does it have if $x_1$, $x_2$, $x_3$ and $x_4$ are positive integers?
\item How many solutions does it have if all $x_i$ are integers and $x_i \geq 2$ for all $i$?
\item How many solutions does it have if all $x_i$ are integers and $x_i \geq i$ for all $i$?
\end{enumerate}
\end{exercise}


\begin{exercise}

    \item How many different combinations of scores are possible if at least one dice shows $6$?

\end{exercise}

\section{Day 6. Generating functions}

\begin{idea}
    Any sequence of numbers 
    \[
    (a_0, a_1, a_2, a_3, \dots)
    \] 
    can be written as a \emph{generating function}
    \[
    g(t) = a_0 + a_1 t^1 + a_2 t^2 + a_3 t^3 + \dots
    \]

    Other variants of a generating function may be also useful 
    \[
    m(t) = a_0 + a_1 \exp(t) + a_2 \exp(2t) + a_3 \exp(3t) + \dots
    \]
    or 
    \[
    e(t) = a_0 + a_1 t^1/1! + a_2 t^2/2! + a_3 t^3 / 3! + \dots
    \]
    or even 
    \[
    f(t) = a_0 + a_1 \cos(t) + a_2 \cos(2t) + a_3 \cos(3t) + \dots
    \]
\end{idea}


\begin{definition}
    We will denote by $\coef(g(t), t^k)$ the coefficient $a_k$ before the term $t^k$ in the expansion
    \[
    g(t) =  a_0 + a_1 t^1 + a_2 t^2 + a_3 t^3 + \dots + a_k t^k + \dots
    \]
\end{definition}

\begin{idea}
    How to discover the equation for the generating function?   
    \begin{itemize}
        \item Represent the studied objects as vertices of a tree. 
        Here $a_n$ will be the number of objects with distance from the root equal to $n$.
        \item Encode objects as words, where $a_n$ is the number of $n$-letter words. 
        These words correspond to path from the root to the object. 
        \item Start from recurrence equation for $a_n$ and use \emph{sigma-boy} approach. 
        \item Start from recurrence equation for $a_n$ and use \emph{two-way ticket} approach. 
        \item Apply first step approach to generating function in probability theory. 
    \end{itemize}
\end{idea}

\section{Day 10. Polya-Redfield counting}

\begin{definition}
    \emph{Necklace} is a cirle of $n$ coloured beads that are considered equivalent under rotations. 

    \emph{Cyclic group} $C_n$ is a collection of all possible rotations of a necklace. 
\end{definition}

\begin{definition}
    \emph{Bracelet} is a cirle of $n$ coloured beads that are considered equivalent under rotations and flips (turn-overs). 

    \emph{Dihedral group} $D_n$ is a collection of all possible rotations and flips of a bracelet.
\end{definition}


We will work with a set $A$ with $\size(A) = n$. 
The number of colours will be denoted as $k$. 
We will denote the set of all colorings of $A$ by $X$. 
The group $G$ acts on $A$ and therefore on $X$ by permuting the elements.

\begin{definition}
    A \emph{group of actions} $G$ should satisfie two properties:
    \begin{enumerate}
        \item For every action $g$ there should exist an anti-action denoted by $g^{-1}$.
        \item For every two actions $a$ and $b$ there should exist the combined action $b\cdot a$. 
        Here $a$ is applied first, and $b$ is applied second. Convention is similar to $\cos \sin(x)$ where $\sin$-function is applied first and then $\cos$-function is applied.
    \end{enumerate}
    From these two principles it follows that every group $G$ contains a «chill» action, identity element, denoted by $e$. 
\end{definition}

\begin{definition}
    Consider an action $g = (12)(3)(4)(56)(789)$. The \emph{cyclic monomial} of the action $g$ is given by
    \[
    \CM_g(x_1, x_2, \dots, x_n) = x_2 \cdot x_1 \cdot x_1 \cdot x_2 \cdot x_3 = x_1^2 x_2^2 x_3.
    \]
    In general the cyclic monomial of an action is the product 
    \[
    \CM_g(x_1, x_2, \dots, x_n) = \prod_{i=1}^n x_i^{r_i}.
    \]
    where $r_i$ is the number of cycles of lenght $i$ in the action $g$.
\end{definition}

\begin{definition}
    The \emph{cyclic index} of the group $G$ is the polynomial given by the scaled sum of cyclic monomials for every action
    \[
    \CI(x_1, x_2, \ldots, x_n) = \frac{1}{\size(G)} \sum_{g\in G} \CM_g(x_1, x_2, \dots, x_n).
    \]
\end{definition}


\begin{theorem}[Polya-Redfield theorem]
    The generating function for all colorings of the set $A$ into three colors ($a$, $b$ and $c$) different with respect to group $G$ is given by
    \[
    g(a, b, c) = \CI(a + b + c, a^2 + b^2 + c^2, a^3 + b^3 + c^3, \dots, a^n + b^n + c^n).
    \]
    %For $k$ colors the formula is 
    %\[
    %g(a_1, a_2, \dots, a_k) = \CI(a_1 + a_2 + \dots + a_k, a_1^2 + a_2^2 + \dots + a_k^2, a_1^3 + a_2^3 + \dots + a_k^3, \dots, a_1^n + a_2^n + \dots + a_k^n).
    %\]
\end{theorem}

\begin{theorem}[Burnside lemma]
   The number of colorings of the set $A$ into $k$ colors different with respect to group $G$  is given by
    \[
    \size(X/G) = g(1, 1, \dots, 1) = \CI(k, k, k, \dots, k) =  \frac{1}{\size(G)} \sum_{g\in G}\size(\fix(g, X)).
    \]
\end{theorem}
