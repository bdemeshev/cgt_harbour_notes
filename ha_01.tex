\subsection*{Home assignment: day 1 $\to$ day 4}

\begin{enumerate}
    \item Simplify
\[
\sum_{a + b + c = 100} {100 \choose a} \cdot {100-a \choose b}.
\]
    \item 1.9(2) in the old enumeration:

    Find explicit formula for the sum 
    \[
    1^3 + 2^3 + \dots + n^3.
    \]
    Hint: represent the general term $k^3$ as the sum of binomial coefficients and then use the hockey stick or Christmas stocking or boomerang!

    \item  {[*]} Using hockey stick decomposition find the sums
    \[
    \sum_{k=1}^{n} {5k^2 + 4k}.
    \]

    \item {[*]} Take a deep breath and calculate 
\[
{-1 \choose 3}, \quad {-5 \choose 5},\quad  {1.5 \choose 3}, \quad  
{-1/2 \choose 4}, \quad  {1/3 \choose k}, \quad  {-1/2 \choose n}. 
\]

    \item Write the Taylor expansion at $x = 0$ for 
\[
\frac{1}{1 - x}, \quad  (1 + x)^{-5}, \quad  (1 + x)^{-1/2}, \quad  
(1 + 3x)^{1/3}, \quad  (1 - 5x)^{1/2}.
\]

    \item Watch a Veritasium video «Newton changed the game»:

    \url{https://youtu.be/gMlf1ELvRzc?si=VASoVKQj3Pk-wstA}.
\end{enumerate}
